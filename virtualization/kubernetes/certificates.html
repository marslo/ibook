extend X509v3 Subject Alternative Name in apiserver.crt

$ ssh master02

//生成2048位的密钥对
$ openssl genrsa -out apiserver-master02.key 2048

$ sudo openssl req -new \
                   -key apiserver-master02.key \
                   -subj "/CN=kube-apiserver," \
                   -out apiserver-master02.csr

// 编辑apiserver-master02.ext文件，内容如下：
$ cat apiserver-master02.ext

              -nodes \
              -out marslo.csr \

              -out marslo.csr

    $ openssl genrsa -out marslo.key 2048

signing the certificate via ca.crt

$ sudo openssl x509 -req \
                    -in marslo.csr \
                    -CA /etc/kubernetes/pki/ca.crt \
                    -CAkey /etc/kubernetes/pki/ca.key \
                    -CAcreateserial \
                    -out marslo.crt \
                    -days 365 \
                    -sha256
