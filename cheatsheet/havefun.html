Simulate type mechine [Pretty Cool!!]

$ sudo apt-get intall pv
$ echo "Very very very very very long words" | pv -qL $[10+(-2 + RANDOM%5)]

# or
$ sudo apt-get install randtype
$ echo "Very very very very very long words" | randtype -m 4
